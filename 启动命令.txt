# =====================================================
# XJ-GPT 协议注册机 · 常用启动命令
# =====================================================

# -----------------------------------------------------
# Web GUI(推荐,带可视化界面 + 实时日志)
# -----------------------------------------------------

# 默认启动,浏览器打开 http://127.0.0.1:5000
python web_app.py

# 换端口
python web_app.py --port 8000

# 暴露到局域网(同 Wi-Fi 的其他机器能访问,自担风险)
python web_app.py --host 0.0.0.0 --port 5000

# 开 DEBUG 日志(看更详细的步骤)
python web_app.py --verbose


# -----------------------------------------------------
# CLI(命令行批量,无界面)
# -----------------------------------------------------

# 单个注册测试
python main.py -n 1

# 批量 10 个,3 并发,单个失败继续
python main.py -n 10 --workers 3 --continue-on-fail

# 批量 5 个串行,每个间隔 30 秒
python main.py -n 5 --delay 30

# 详细日志模式
python main.py -n 1 --verbose
